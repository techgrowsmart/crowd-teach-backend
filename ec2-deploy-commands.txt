# EC2 Deployment Commands
# Run these commands in order on your EC2 server

## 1. Connect to EC2 Server
ssh -i your-key.pem ec2-user@your-ec2-ip

## 2. Navigate to backend directory
cd /home/ec2-user/crowd-teach-gogrowsmart-backend

## 3. Pull latest changes
git pull origin main

## 4. Make deployment script executable
chmod +x deploy-ec2.sh

## 5. Run deployment script
./deploy-ec2.sh

## 6. Alternative manual deployment:
# Update system
sudo yum update -y

# Install Docker
sudo yum install -y docker
sudo service docker start
sudo usermod -a -G docker ec2-user

# Install Docker Compose
sudo curl -L "https://github.com/docker/compose/releases/latest/download/docker-compose-$(uname -s)-$(uname -m)" -o /usr/local/bin/docker-compose
sudo chmod +x /usr/local/bin/docker-compose

# Stop existing services
docker-compose down

# Build and start services
docker-compose up --build -d

# Check status
docker-compose ps

# View logs
docker-compose logs -f

## 7. Verify deployment
curl http://localhost:3000
curl http://localhost:3000/api/auth/login

## 8. Check server logs
docker-compose logs app

## 9. If you need to restart services
docker-compose restart
docker-compose restart app

## 10. Check MongoDB and Redis connectivity
docker-compose exec app node -e "require('./config/mongoDB')()"
docker-compose exec app node -e "require('./config/redis').connect()"
